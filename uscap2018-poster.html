library(ggplot2)

g1 <- DF %>% 
    select(p53v:p53d) %>% 
    gather(marker, value, p53v:p53d) %>% 
    ggplot(aes(x = marker, y = value)) +
    stat_summary(fun.data = "mean_cl_boot", color = "red", size = 1.5) +
    geom_point() +

    labs(x = "", y = "% positive cells", title = "A")
