# Diagrama de dispersión
g2_cor <- function(x, y) {
    DF <- data.frame(x, y)
    ggplot(DF, aes(x, y)) +
    geom_jitter(color = "steelblue") +
    geom_smooth(linetype = 3) +
    labs(x = "Inspección visual", y = "Evaluación digital") +
    theme_bw()
}
